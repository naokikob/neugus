r=91 <-  x<=101, mc91(x, r)
mc91(x, x-10) <- x>100
mc91(x,r) <- x<=100, mc91(x, s), mc91(s, r)

data:

negative:
 (101, 90), ...,
  (* (x, r), x<=101,  r<>91 のものすべて *)

positive:
 (101, 91), (102, 92), ...
  (* (x, x-10), x>=101のものすべて *)

implications:
 {(101, 91), (91, 91)=> (101,91)}, ...


===================
simple example:


 false <- p(4)
 p(5) <- true
 p(x) <- p(x-1).

(4, 0)
(5, 1)

(-25,1)->(-26,1)
(1,1)->(2,1)
...
(24,1)->(25,1)
